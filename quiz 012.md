12. Deduce the code using the input and output provided

#code

    input1=int(input("Input a num: "))
    input2=int(input("Input another num: "))
    def mystery(input1,input2):
        minus=input1-1
        mult=minus*input2
        sum=mult+input1
        return sum
    out=mystery(input1,input2)
    print(out)
